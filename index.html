import pygame import sys import math import random # Initialize game engine pygame.init() # 1. SCREEN SETUP WIDTH, HEIGHT = 1000, 700 screen = pygame.display.set_available_cameras screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Coding Power: Epic Boss Fight") clock = pygame.time.Clock() # COLORS (RGB) BLACK = (20, 20, 30) WHITE = (255, 255, 255) BLUE = (0, 191, 255) RED = (255, 50, 50) PURPLE = (147, 112, 219) YELLOW = (255, 215, 0) GREEN = (50, 205, 50) # 2. GAME VARIABLES & CLASSES class Player: def __init__(self): self.rect = pygame.Rect(100, HEIGHT // 2, 40, 40) self.speed = 6 self.health = 100 self.max_health = 100 def move(self, keys): if keys[pygame.K_w] or keys[pygame.K_UP]: self.rect.y -= self.speed if keys[pygame.K_s] or keys[pygame.K_DOWN]: self.rect.y += self.speed if keys[pygame.K_a] or keys[pygame.K_LEFT]: self.rect.x -= self.speed if keys[pygame.K_d] or keys[pygame.K_RIGHT]: self.rect.x += self.speed # Keep player on screen self.rect.clamp_ip(pygame.Rect(0, 0, WIDTH, HEIGHT)) class Bullet: def __init__(self, x, y, dx, dy, color=BLUE): self.x = x self.y = y self.dx = dx self.dy = dy self.radius = 6 self.color = color def update(self): self.x += self.dx self.y += self.dy def draw(self, surface): pygame.draw.circle(surface, self.color, (int(self.x), int(self.y)), self.radius) class Boss: def __init__(self): self.rect = pygame.Rect(WIDTH - 150, HEIGHT // 2 - 60, 100, 120) self.health = 500 self.max_health = 500 self.phase = 1 self.move_dir = 1 self.shoot_cooldown = 0 def update(self, player_y): # Phase 1: Simple vertical floating up and down if self.phase == 1: self.rect.y += self.move_dir * 3 if self.rect.top <= 50 or self.rect.bottom >= HEIGHT - 50: self.move_dir *= -1 if self.health <= self.max_health / 2: self.phase = 2 # Trigger Phase 2! # Phase 2: Aggressive tracking of player's Y level elif self.phase == 2: if self.rect.centery < player_y: self.rect.y += 5 elif self.rect.centery > player_y: self.rect.y -= 5 # 3. INSTANTIATE OBJECTS player = Player() boss = Boss() player_bullets = [] boss_bullets = [] # UI Fonts font = pygame.font.SysFont("Arial", 24, bold=True) large_font = pygame.font.SysFont("Arial", 60, bold=True) # 4. MAIN GAME LOOP running = True game_over = False victory = False while running: # Event handling for event in pygame.event.get(): if event.type == pygame.QUIT: running = False pygame.quit() sys.exit() # Player shoots using SPACEBAR if event.type == pygame.KEYDOWN and not game_over and not victory: if event.key == pygame.K_SPACE: # Shoot forward bullet player_bullets.append(Bullet(player.rect.right, player.rect.centery, 12, 0, BLUE)) if not game_over and not victory: # Get movement input keys = pygame.key.get_pressed() player.move(keys) boss.update(player.rect.centery) # Boss Shooting Logic boss.shoot_cooldown += 1 cooldown_target = 25 if boss.phase == 1 else 12 # Shoots twice as fast in Phase 2 if boss.shoot_cooldown >= cooldown_target: boss.shoot_cooldown = 0 if boss.phase == 1: # Direct straight shots boss_bullets.append(Bullet(boss.rect.left, boss.rect.centery, -8, 0, RED)) else: # Phase 2 Target tracking shots: calculate angle to player angle = math.atan2(player.rect.centery - boss.rect.centery, player.rect.centerx - boss.rect.centerx) dx = math.cos(angle) * 9 dy = math.sin(angle) * 9 boss_bullets.append(Bullet(boss.rect.left, boss.rect.centery, dx, dy, PURPLE)) # Update Player Bullets for b in player_bullets[:]: b.update() if b.x > WIDTH: player_bullets.remove(b) elif boss.rect.collidepoint(b.x, b.y): boss.health -= 10 player_bullets.remove(b) if boss.health <= 0: victory = True # Update Boss Bullets for b in boss_bullets[:]: b.update() if b.x < 0 or b.y < 0 or b.y > HEIGHT: boss_bullets.remove(b) elif player.rect.collidepoint(b.x, b.y): player.health -= 15 boss_bullets.remove(b) if player.health <= 0: game_over = True # 5. DRAWING EVERYTHING ON SCREEN screen.fill(BLACK) if not game_over and not victory: # Draw Entities pygame.draw.rect(screen, BLUE, player.rect) boss_color = RED if boss.phase == 1 else PURPLE pygame.draw.rect(screen, boss_color, boss.rect) # Draw Bullets for b in player_bullets: b.draw(screen) for b in boss_bullets: b.draw(screen) # Draw HUD / Health Bars # Player Health Bar pygame.draw.rect(screen, RED, (20, 20, 200, 20)) pygame.draw.rect(screen, GREEN, (20, 20, int(200 * (player.health / player.max_health)), 20)) p_text = font.render(f"PLAYER HEALTH: {max(0, player.health)}%", True, WHITE) screen.blit(p_text, (20, 45)) # Boss Health Bar pygame.draw.rect(screen, WHITE, (WIDTH - 420, 20, 400, 25)) pygame.draw.rect(screen, boss_color, (WIDTH - 420, 20, int(400 * (boss.health / boss.max_health)), 25)) b_text = font.render(f"BOSS PHASE {boss.phase}", True, WHITE) screen.blit(b_text, (WIDTH - 420, 50)) # End Screen States elif game_over: msg = large_font.render("GAME OVER", True, RED) screen.blit(msg, (WIDTH // 2 - 170, HEIGHT // 2 - 50)) elif victory: msg = large_font.render("BOSS DESTROYED!", True, YELLOW) screen.blit(msg, (WIDTH // 2 - 250, HEIGHT // 2 - 50)) pygame.display.flip() clock.tick(60) # Lock game to 60 FPS
