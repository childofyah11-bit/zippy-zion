/** * SMALL TOWN STAR (GOOFBALL) - CHRONO AUDIO ENGINE * System Architecture Block 1/3: Engine & Panel Setup */ let audioSource, fft, fontCode; let goofballs = []; let matrixDrops = []; let kineticLyrics = []; let activeLyricIdx = 0; // Global Visual State Configurations let sysState = { bgHue: '#020208', particleCap: 200, physicsSpeed: 3.5, glitchGain: 0.4, matrixFeedSpeed: 4, colorScheme: 'Hyper Neon Cyan', gainMultiplier: 1.8, glowDepth: 15, gridGeometry: true, waveProfile: 4 }; // Interface Controls Layout let gui = {}; let lyricTimeline = [ "Goofball bouncing-off-the-wall-ball", "Everybody knows my name", "Staying in my own lane", "Dancing in the quiet rain", "Screen bright, App tapped", "Beats play... Soundtrack for the day", "Speakers loud, Top speed", "Deep rhythm takes control", "Lights fade, Stage set", "God made me perfectly after all 💫" ]; function setup() { createCanvas(windowWidth - 280, windowHeight); fft = new p5.FFT(0.85, 512); // Audio fallback initialization audioSource = new p5.AudioIn(); audioSource.start(); fft.setInput(audioSource); buildControlCenter(); syncEngineArrays(); } function windowResized() { resizeCanvas(windowWidth - 280, windowHeight); } /** * System Architecture Block 2/3: Graphic Rendering Engine */ function draw() { background(sysState.bgHue); // Frequency Spectrum Distribution fft.analyze(); let subBass = fft.getEnergy("bass") * sysState.gainMultiplier; let midRange = fft.getEnergy("mid") * sysState.gainMultiplier; let transients = fft.getEnergy("treble") * sysState.gainMultiplier; updateEngineState(); // Layer 1: Holographic Tech Grid Background if (sysState.gridGeometry) { drawHoloGrid(midRange); drawMatrixTextStream(subBass); } // Layer 2: Core Reactive Waveform (Oscilloscope Orbit) drawSpatialWaveform(subBass, midRange); // Layer 3: Kinetic Particles ("Goofballs") renderGoofballPhysics(subBass, transients); // Layer 4: Graphic Overlay & Real-time Text Systems renderKineticLyrics(subBass); applyGlitchVignette(transients); } function drawHoloGrid(mid) { stroke(getThemeColor(40)); strokeWeight(0.5); let step = 40 + (mid * 0.05); for (let x = 0; x < width; x += step) line(x, 0, x, height); for (let y = 0; y < height; y += step) line(0, y, width, y); } function drawMatrixTextStream(bass) { fill(getThemeColor(120)); textSize(11); textFont('Courier New'); if (random(1) < 0.25) { matrixDrops.push({ x: random(width), y: 0, v: random(1, 5) * (sysState.matrixFeedSpeed * 0.5), token: random(['01', 'A+', 'SYS_OK', '{}', '[]', 'GRACE', 'STARS']) }); } for (let i = matrixDrops.length - 1; i >= 0; i--) { let d = matrixDrops[i]; text(d.token, d.x, d.y); d.y += d.v + (bass * 0.03); if (d.y > height) matrixDrops.splice(i, 1); } } function drawSpatialWaveform(bass, mid) { push(); translate(width / 2, height / 2); noFill(); // Chromatic Aberration Simulation Effect (Double Ring) strokeWeight(sysState.waveProfile); stroke(getThemeColor(220)); renderWaveRing(1.0, bass); strokeWeight(sysState.waveProfile * 0.5); stroke(255, 50, 100, 150); // Red fringe glitch ring renderWaveRing(1.05, mid); pop(); } function renderWaveRing(scaleF, energy) { let waveform = fft.waveform(); beginShape(); for (let i = 0; i < waveform.length; i += 4) { let angle = map(i, 0, waveform.length, 0, TWO_PI); let offset = map(waveform[i], -1, 1, -60, 60) * (energy * 0.02); let r = (160 + offset) * scaleF; let x = r * cos(angle); let y = r * sin(angle); vertex(x, y); } endShape(CLOSE); } /** * System Architecture Block 3/3: Physics & Interface Bindings */ class GoofballParticle { constructor() { this.pos = createVector(random(width), random(height)); this.vel = p5.Vector.random2D().mult(random(1, 3)); this.dim = random(20, 50); } motionEngine(bass, treble) { let dynamicSpeed = sysState.physicsSpeed * (1 + bass * 0.008); this.pos.add(p5.Vector.mult(this.vel, dynamicSpeed * 0.3)); // Bounds Check with Wall Bounce Elasticity if (this.pos.x < 0 || this.pos.x > width) this.vel.x *= -1; if (this.pos.y < 0 || this.pos.y > height) this.vel.y *= -1; this.currentDim = this.dim + (treble * 0.15); } render() { push(); drawingContext.shadowBlur = sysState.glowDepth; drawingContext.shadowColor = getThemeColor(255); noStroke(); fill(getThemeColor(180)); ellipse(this.pos.x, this.pos.y, this.currentDim); // Internal Core Label fill(sysState.bgHue); textAlign(CENTER, CENTER); textSize(this.currentDim * 0.28); textFont('sans-serif'); text("GOOF", this.pos.x, this.pos.y); pop(); } } function renderKineticLyrics(bass) { // Rotate lyric on structural bass drops if (bass > 210 && random(1) < 0.05) { activeLyricIdx = (activeLyricIdx + 1) % lyricTimeline.length; } push(); textAlign(CENTER); textFont('Courier New'); textSize(24 + (bass * 0.05)); fontWeight(BOLD); fill(255); text(lyricTimeline[activeLyricIdx], width / 2, height - 60); pop(); } function applyGlitchVignette(treble) { if (treble > 190 && random(1) < sysState.glitchGain) { stroke(255, 0, 100, 200); strokeWeight(random(2, 8)); line(0, random(height), width, random(height)); fill(getThemeColor(25)); rect(random(width), random(height), random(50, 200), random(10, 40)); } } function buildControlCenter() { let menuX = width + 25; let panel = createDiv('').position(menuX - 10, 0).size(280, height).style('background', '#0b0c16').style('border-left', '2px solid #222540').style('padding', '15px'); let header = createDiv('🎛️ CORE MATRIX COMMAND').parent(panel).style('color', '#00ffcc').style('font-family', 'monospace').style('font-weight', 'bold').style('margin-bottom', '20px'); gui.audioBtn = createButton('⚡ INIT MIC / AUDIO DRIVER').parent(panel).style('width', '100%').style('background', '#16192b').style('color', '#00ffcc').style('border', '1px solid #00ffcc').style('padding', '8px').style('cursor', 'pointer'); gui.audioBtn.mousePressed(() => userStartAudio()); createGuiLabel('THEME COLOR SELECTOR', panel); gui.themeOpt = createSelect().parent(panel).style('width', '100%').style('background', '#16192b').style('color', '#fff'); gui.themeOpt.option('Hyper Neon Cyan'); gui.themeOpt.option('Cyberpunk Crimson'); gui.themeOpt.option('Liquid Gold'); createGuiLabel('PARTICLE CAP (GOOFBALLS)', panel); gui.capSld = createSlider(20, 400, 200, 1).parent(panel); createGuiLabel('VELOCITY VECTOR AMPLITUDE', panel); gui.velSld = createSlider(0.5, 10.0, 3.5, 0.1).parent(panel); createGuiLabel('GLITCH MATRIX TRANSVERTER', panel); gui.gliSld = createSlider(0, 1, 0.4, 0.05).parent(panel); createGuiLabel('CODE FIELD SPEED', panel); gui.matSld = createSlider(0, 12, 4, 0.5).parent(panel); createGuiLabel('AUDIO AMPLITUDE SENSITIVITY', panel); gui.gainSld = createSlider(0.5, 4.0, 1.8, 0.1).parent(panel); } function updateEngineState() { sysState.particleCap = gui.capSld.value(); sysState.physicsSpeed = gui.velSld.value(); sysState.glitchGain = gui.gliSld.value(); sysState.matrixFeedSpeed = gui.matSld.value(); sysState.gainMultiplier = gui.gainSld.value(); sysState.colorScheme = gui.themeOpt.value(); syncEngineArrays(); } function syncEngineArrays() { while (goofballs.length < sysState.particleCap) goofballs.push(new GoofballParticle()); while (goofballs.length > sysState.particleCap) goofballs.pop(); } function renderGoofballPhysics(bass, treble) { for (let ball of goofballs) { ball.motionEngine(bass, treble); ball.render(); } } function createGuiLabel(labelString, parentPanel) { createDiv(labelString).parent(parentPanel).style('color', '#6b7280').style('font-family', 'sans-serif').style('font-size', '10px').style('margin-top', '18px').style('margin-bottom', '4px').style('letter-spacing', '1px'); } function getThemeColor(alpha) { if (sysState.colorScheme === 'Hyper Neon Cyan') return color(0, 255, 204, alpha); if (sysState.colorScheme === 'Cyberpunk Crimson') return color(255, 0, 90, alpha); if (sysState.colorScheme === 'Liquid Gold') return color(255, 210, 0, alpha); return color(255, alpha); }
